int ldr;  

void setup() 

{
  pinMode(13, OUTPUT); 
  Serial.begin(9600);   
   ldr = analogRead(A0); 
}

void loop() 
{
 

  if (ldr < 200)
  {      
    Serial.println("Darkness here, turning LED on");
    digitalWrite(13, HIGH);  
  } 
  else 
  {             
    Serial.println("Sufficient Light");
    digitalWrite(13, LOW);  
  }
  
  delay(500); 
}
