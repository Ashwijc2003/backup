const int lm35_pin = A1;  // LM35 sensor connected to analog pin A1

void setup() {
  Serial.begin(9600);  // Initialize serial communication
}

void loop() {
  int temp_adc_val;
  float temp_val;
  
  temp_adc_val = analogRead(lm35_pin);  // Read analog value
  temp_val = temp_adc_val * 4.882;      // Convert to millivolts (5V/1024 = 4.882mV per unit)
  temp_val = temp_val / 10;             // Convert to Celsius (10mV per degree)
  
  Serial.print("Temperature: ");
  Serial.print(temp_val);
  Serial.println(" Degree Celsius");
  
  delay(2000);  // Wait 2 seconds between readings
}
