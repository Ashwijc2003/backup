#include <LiquidCrystal.h>


const int rs = 12, en = 13, d4 = 8, d5 = 9, d6 = 10, d7 = 11;


LiquidCrystal lcd(rs, en, d4, d5, d6, d7);

void setup() {
  lcd.begin(16, 2);         
  lcd.clear();               
  lcd.setCursor(0, 0);       
  lcd.print("HELLO 10T");  
  delay(1000);          
}

void loop() {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("MVJCE");

  lcd.setCursor(0, 1);
  lcd.print("BI ALL");     

  delay(900);

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("BOYS GIRLS");

  delay(900);
}
