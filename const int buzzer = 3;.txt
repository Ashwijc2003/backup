const int buzzer = 3;  
const int sensor = 4;  
int state;             

void setup() {
  pinMode(sensor, INPUT);   
  pinMode(buzzer, OUTPUT);  
}

void loop() {
  state = digitalRead(sensor); 
  
  if (state == HIGH) {
    tone(buzzer, 400);  
  } else {
    noTone(buzzer);     
  }
  
  delay(2000);  
}
